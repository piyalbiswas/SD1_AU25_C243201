import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int X = sc.nextInt();
        int Y = sc.nextInt();
        int Z = sc.nextInt();

        int totalStudents = X * Y;
        if (Z > totalStudents / 2.0) {
            System.out.println("YES");
        } else {
            System.out.println("NO");
        }
    }
}
