import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int X = sc.nextInt();
        int Y = sc.nextInt();
        
        int totalHours = 5 * X - X + Y; // 5 days total, subtract one X for Wednesday and add Y
        System.out.println(totalHours);
    }
}
