import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int X = sc.nextInt();
        int finishTime = X + 3;
        if (finishTime <= 22) {
            System.out.println("YES");
        } else {
            System.out.println("NO");
        }
    }
}
